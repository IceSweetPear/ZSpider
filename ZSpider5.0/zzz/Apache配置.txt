绑定域名
	进入apache/conf/httpd.conf
	
	<VirtualHost *:80>
	    ServerAdmin webmaster@dummy-host.localhost
	    DocumentRoot "D:/EmpireServer/web" (此处意思为，你对应解析的域名的代码位置)
	    ServerName aaa.com (此处地址与你在hosts文件中写的一样)
	    ServerAlias *.aaa.com ( 识别解析生成的域名)
	    ErrorLog "logs/localhost-error_log"
	</VirtualHost>

命令
	启动
		apache2/bin/apachectl start apache
	停止
		apache2/bin/apachectl stop
	重启
		apache2/bin/apachectl restart

添加为linux 启动目录
	cp /usr/local/httpd/bin/apachectl /etc/rc.d/init.d/httpd	复制软链 可以使用服务命令 service httpd start
注册为linux 服务
	#chkconfig:345 85 15
	#description:Start and stop the Apache HTTP Server 
	chkconfig --add httpd
