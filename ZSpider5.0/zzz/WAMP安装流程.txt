1,安装成功 Apache PHP

2,  httpd.conf  PHP安装成Apache的一个功能模块
	LoadModule php7_module 'C:/zzz/php/php7apache2_2.dll'

3,  httpd.conf  识别 .php 的文件 并处理
	AddType application/x-httpd-php .php

4,  httpd.conf  加载 php 虚拟主机配置文件
	Include conf/extra/httpd-vhosts.conf

5,  httpd.conf  加载 php ini 配置文件
	PHPIniDir  C:/zzz/php

6  php.ini  修改时区
	date.timezone = PRC

7  php.ini  加载mysql扩展
	extension=php_mysql.dll




Ubantu

安装 apache
	wget httpd-2.4.2.tar.bz2
	tar -xvf httpd-2.4.2.tar.bz2
	cd httpd-2.4.2

	wget tar cd apr   
	sudo ./configure
	sudo  make
	sudo make install
	
	wget tar cd apr-util    
	sudo ./configure
	sudo make
	sudo make install

	wget tar cd pcre1
	sudo ./configure --prefix=/usr/local/pcre  
	sudo make  
	sudo make install

	./configure --prefix=/usr/local/apache --enable-module=shared  --with-apr=/usr/local/apr/ --with-apr-util=/usr/local/apr-util/ --with-pcre=/usr/local/pcre
	sudo make
	sudo make install 
	
	/usr/local/apache/bin/apachectl -k start
	/usr/local/apache/bin/apachectl -k restart
	/usr/local/apache/bin/apachectl -k stop
	
	cp /usr/local/apache/bin/apachectl   /usr/sbin/    
	vim /exc/environment
	source /exc/environment 

加载php
	LoadModule php7_module modules/libphp7.so
	ServerName localhost:80
	DirectoryIndex index.html index.php

	sudo apt-get Cpurge remove apache2     卸载 apache2
	sudo apt-get Cpurge remove apache2-utils   卸载 apache2


加载 pcntl 扩展
	cd  php/ext/pcntl
	phpize
	./configure --with-php-config=/usr/local/php/bin/php-config
	make
	sudo make install
	vim  /etc/php.ini
	extension=php/ext/pcntl/moudle/pcntl.so
加载 redis
	wget redis
	cd redis
	sudo make
	sudo make install
	vim  ../redis.conf    daemonize no=>daemonize yes
	redis-server  ../redis.conf
	ps -ef|grep redis
加载phpredis扩展
	wget phpredis
	cd /etc/phpredis
	phpize
	./configure
	make
	make install
	vim php.ini
	extension=/phpredis/modules/redis.so
